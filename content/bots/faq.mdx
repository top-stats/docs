import { Accordion, Accordions } from 'fumadocs-ui/components/accordion';

<Accordions type='single'>
  <Accordion id='what-is-top-stats' title='What is Top Stats?'>
    Top Stats, previously DBL Statistics, is a analytical site that provides statistics on your favourite Discord Bots
  </Accordion>
  <Accordion id='where-does-top-stats-get-its-data' title='Where does Top Stats get its data?'>
    We source our Discord bot data from Top.gg, every hour.
  </Accordion>
  <Accordion id='remove-wrong-data' title='I posted the wrong data to Top.gg and its shown on your site, can I remove this data point?'>
    If you would like to remove a certain data point in your data, you can do so by contacting us in our [Discord Server](https://discord.gg/VDf8CG6KGA).
  </Accordion>
  <Accordion id='why-not-showing' title='Why is my bot not showing up?'>
    This may be because your bot has not yet been approved, or has recently been approved. Please wait till the next hour for it to pop up.
  </Accordion>
  <Accordion id='data-incorrect' title='Why is my bots data incorrect or not showing up?'>
    In order for your data to appear on Top Stats, you will need to post your server count and shard count to [Top.gg](https://top.gg) through their API.
  </Accordion>
  <Accordion id='data-issue' title='There is an issue with my data or a graph isnt showing what Id expect.'>
    Please contact our team on our [Discord Server](https://discord.gg/VDf8CG6KGA), so that we can assist you with this.
  </Accordion>
  <Accordion id='request-data' title='Can I request all my data?'>
    At this moment we don't have a way for you to do this. We may in the future but until then unfortunately there is no way to do this.
  </Accordion>
  <Accordion id='delete-data' title='How can I delete all of my data?'>
    Please contact our team at privacy@topstats.gg, and we will remove your personal data. Please note that if you log back into our site after account deletion, we will store your data once more
  </Accordion>
  <Accordion id='affiliation' title='Are we affiliated with Top.gg?'>
    No, Top Stats is an independent entity and we are not affiliated with [Top.gg](https://top.gg).
  </Accordion>
  <Accordion id='change-time-format' title='Can I change the time format?'>
    Topstats always adjusts the time format to the language set in your browser. Change the language in your browser to change the time format.
  </Accordion>
</Accordions>
